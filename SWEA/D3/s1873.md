package day0204;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class D1873_battleField {
	public static char[][] field;		//í•„ë“œ ì •ë³´
	public static int width, height;	//í•„ë“œ í¬ê¸°
	public static int now_w, now_h;		//ì „ì°¨ì˜ ìœ„ì¹˜
	
	/* ëª…ë ¹ì–´ ìˆ˜í–‰ ë©”ì„œë“œ
	 * - U,D,L,R : ì´ë™í•  ë¸”ë¡ì´ í‰ì§€ì¼ ê²½ìš° ì´ë™. ì…ë ¥ëœ ëª…ë ¹ì–´ì— ë§ê²Œ ì „ì°¨ ë°©í–¥ ì¡°ì ˆ

	public static void command(char c) {
		switch(c) {
		case 'U': 
			if(now_h>0 && field[now_h-1][now_w] == '.') {
				field[now_h][now_w] = '.';
				now_h--;
			}
			field[now_h][now_w] = '^';
			break;
			
		case 'D': 
			if(now_h<height-1 && field[now_h+1][now_w] == '.') {
				field[now_h][now_w] = '.';
				now_h++;
			}
			field[now_h][now_w] = 'v';
			break;
			
		case 'L': 
			if(now_w>0 && field[now_h][now_w-1] == '.') {
				field[now_h][now_w] = '.';
				now_w--;
			}
			field[now_h][now_w] = '<';
			break;
			
		case 'R': 
			if(now_w<width-1 && field[now_h][now_w+1] == '.') {
				field[now_h][now_w] = '.';
				now_w++;
			}
			field[now_h][now_w] = '>';
			break;

		case 'S':
			switch(field[now_h][now_w]) {
			case '^':
				for(int i=now_h-1; i>=0; i--) {
					if(seekWall(i, now_w)) break;
				}
				break;
			case 'v':
				for(int i=now_h+1; i<height; i++) {
					if(seekWall(i, now_w)) break;
				}
				break;
			case '<':
				for(int j=now_w-1; j>=0; j--) {
					if(seekWall(now_h, j)) break;
				}
				break;
			case '>':
				for(int j=now_w+1; j<width; j++) {
					if(seekWall(now_h, j)) break;
				}
				break;
			}
			
		}
		
	}
	
	/* ë²½ì„ ì°¾ëŠ” ë©”ì„œë“œ
	 * - í•´ë‹¹ ìœ„ì¹˜ì˜ ë¸”ë¡ì´ ë²½ì¼ ê²½ìš° true, ì•„ë‹ê²½ìš° false ë°˜í™˜
	 */
	public static boolean seekWall(int h, int w) {
		if(field[h][w]=='*') {
			field[h][w] = '.';
			return true;
		}else if(field[h][w]=='#') {
			return true;
		}else {
			return false;
		}
	}
	
	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		int T = Integer.parseInt(br.readLine());
		String line;
		
		for(int t=1; t<=T; t++) {				// 1.í•„ë“œì •ë³´ ì…ë ¥

			for(int i=0; i<height; i++) {
				line = br.readLine();
				for(int j=0; j<width; j++) {
					field[i][j] = line.charAt(j);
					if(field[i][j]=='<'||field[i][j]=='>'||field[i][j]=='^'||field[i][j]=='v') {
						now_h = i;
						now_w = j;
					}
				}
			}
			
			int cmt = Integer.parseInt(br.readLine());	
			line = br.readLine();
			for(int c=0; c<cmt; c++) {			// 2.ëª…ë ¹ì–´ ìˆ˜í–‰
				command(line.charAt(c));
			}
			
			System.out.print("#" + t + " ");	// 3.ì¶œë ¥

					System.out.print(l);
				}
				System.out.println();
			}
			
		}
	}

}
