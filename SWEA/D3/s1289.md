package day0203;

import java.util.Scanner;

public class SW1289_memoryRestore {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int T = sc.nextInt();
		
		for(int t=1; t<=T; t++) {
			String line = sc.next();
			
			char init = '0';
			int cnt = 0;
			for(int i=0; i<line.length(); i++) {
				if(line.charAt(i) != init) {
					cnt++;
					init = line.charAt(i);
				}
			}
			System.out.println("#" + t + " " + cnt);
		}
	}

}
